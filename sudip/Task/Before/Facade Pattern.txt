public class Car {

    private String name;
    private String model;

    public Car() {

    }

    public String getName() {
        return this.name;
    }

    public String SetName(name) {
        this.name = name;
    }

    public String getModel() {
        return this.model;
    }

    public String SetModel(model) {
        this.model = model;
    }
}
